#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

int N;
char **DICT;
char letters[8];
char must;

bool used[7];
char cur[8];
char **results = NULL;
int resCount = 0, resCap = 0;

void sort7(char s[8]) {
    for (int i = 1; i < 7; i++) {
        char key = s[i];
        int j = i - 1;
        while (j >= 0 && s[j] > key) {
            s[j + 1] = s[j];
            j--;
        }
        s[j + 1] = key;
    }
}

int dict_has(const char *w) {
    int lo = 0, hi = N - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        int cmp = strcmp(w, DICT[mid]);
        if (cmp == 0) return 1;
        if (cmp < 0) hi = mid - 1; else lo = mid + 1;
    }
    return 0;
}

void addResult(const char *word) {
    if (resCount == resCap) {
        resCap = resCap ? resCap * 2 : 16;
        results = realloc(results, resCap * sizeof(char*));
    }
    results[resCount] = malloc(strlen(word) + 1);
    strcpy(results[resCount], word);
    resCount++;
}

void backtrack(int depth, int target, int hasMust) {
    if (depth == target) {
        if (hasMust) {
            cur[depth] = '\0';
            if (dict_has(cur)) addResult(cur);
        }
        return;
    }
    for (int i = 0; i < 7; i++) {
        if (used[i]) continue;
        used[i] = true;
        cur[depth] = letters[i];
        backtrack(depth + 1, target, hasMust || (letters[i] == must));
        used[i] = false;
    }
}

int cmpstr(const void *a, const void *b) {
    return strcmp(*(char **)a, *(char **)b);
}

int main(void) {
    char buf[64];
    scanf("%63s", buf);
    must = buf[3];
    for (int i = 0; i < 7; i++) letters[i] = buf[i];
    letters[7] = '\0';
    sort7(letters);

    scanf("%d", &N);
    DICT = malloc(N * sizeof(char*));
    for (int i = 0; i < N; i++) {
        scanf("%63s", buf);
        DICT[i] = malloc(strlen(buf) + 1);
        strcpy(DICT[i], buf);
    }

    for (int target = 4; target <= 7; target++) {
        memset(used, 0, sizeof used);
        backtrack(0, target, 0);
    }

    qsort(results, resCount, sizeof(char*), cmpstr);
    for (int i = 0; i < resCount; i++) {
        if (i == 0 || strcmp(results[i], results[i-1]) != 0) puts(results[i]);
        free(results[i]);
    }

    free(results);
    for (int i = 0; i < N; i++) free(DICT[i]);
    free(DICT);
    return 0;
}
